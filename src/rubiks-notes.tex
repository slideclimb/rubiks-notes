    \subsection{The White Cross}

    \RubikCubeGreyWY

    \RubikSliceTopR{X}{O}{X}{X}{G}{X}

    \begin{goal}

        Note that the colours on the side of the cross should match the colours of the centrepieces.
    \end{goal}
